#tasks file for mynewrole
- name: Installing all the Packages
  yum:
    name: "{{ item }}"
    state: present
  loop: "{{ pkgs }}"
- name: Starting and enabling the services
  service: 
    name: "{{ item }}"
    state: started
    enabled: true
  loop: "{{ pkgs }}"
- name: Create file from template
  template:
    src: index.html.j2
    dest: /var/www/html/index.html
